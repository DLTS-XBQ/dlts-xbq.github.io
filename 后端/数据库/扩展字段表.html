        # 扩展字段表

        扩展字段表存在的意义是为了提供一种灵活的方式来存储和管理额外的数据，这些数据可能不是主要数据表中需要跟踪的标准字段。它可以用于以下情况：

        • 动态添加字段：当业务需求变化时，可能需要在现有表中添加新字段。但是由于数据库设计的限制，这可能会导致数据表结构的改变和数据迁移的复杂性。扩展字段表可以避免这个问题，因为可以动态地向扩展字段表中添加新字段，而不会影响到主要的数据表结构。
        • 存储非结构化数据：有时候，需要存储一些非结构化的数据，例如图片、音频或视频文件等。将这些数据存储在扩展字段表中可以更好地管理和检索这些数据。
        • 扩展元数据：扩展字段表可以用来存储和管理与主要数据表相关的元数据，例如创建时间、修改时间、创建人等。这些元数据可以提供更多关于主要数据的信息和上下文。
        • 灵活性和扩展性：扩展字段表可以为数据库提供更大的灵活性和扩展性。它可以适应不同的业务需求和数据变化，而不需要频繁地修改主要数据表结构。
        • 针对性：特殊业务场景下，可能存在较少的特别数据需要额外的存储字段，一般情况下是需要在主体表加一个字段进行存储，但该列字段的值仅仅只有较少的特别数据有值，所以推荐通过扩展字段表进行扩展，这样做的好处就是主体表更加统一，数据美观，结构紧凑。

        总的来说，扩展字段表存在的意义是为了提供一种更灵活、更可扩展的方法来存储和管理额外的数据，以满足业务需求的变化和数据的复杂性。

        -- 建议表命名格式: 目标表名_ext
create table table_ext
(
    id          bigint                             not null comment '主键id'
        primary key,
    main_id     bigint                             not null comment '主表id',
    field_name  VARCHAR(128)                       not null comment '扩展字段名称',
    field_type  varchar(128)                       not null comment '扩展字段类型',
    field_value varchar(255)                       not null comment '扩展字段值',
    create_id   bigint                             not null comment '创建人id',
    create_time datetime default CURRENT_TIMESTAMP not null comment '创建时间',
    update_id   bigint                             not null comment '修改人id',
    update_time datetime default CURRENT_TIMESTAMP not null on update CURRENT_TIMESTAMP comment '修改时间'
) comment '字段扩展表';

        -- 建议表命名格式: 目标表名_EXT
CREATE TABLE TABLE_EXT
(
    ID          NUMBER(20)                          NOT NULL
        CONSTRAINT TABLE_EXT_PK PRIMARY KEY,
    MAIN_ID     NUMBER(20)                          NOT NULL,
    FIELD_NAME  VARCHAR2(255)                       NOT NULL,
    FIELD_TYPE  VARCHAR2(255)                       NOT NULL,
    FIELD_VALUE VARCHAR2(1024)                      NOT NULL,
    CREATE_ID   NUMBER(20)                          NOT NULL,
    CREATE_TIME TIMESTAMP DEFAULT SYSDATE NOT NULL,
    UPDATE_ID   NUMBER(20)                          NOT NULL,
    UPDATE_TIME TIMESTAMP DEFAULT SYSDATE NOT NULL
);
-- 自动更新时间
create or replace trigger TABLE_EXT_TRIGGER
    before update on TABLE_EXT for each row
begin
    :new.UPDATE_TIME := SYSDATE;
end;
-- 字段注释
comment on table TABLE_EXT is '字段扩展表';
comment on column TABLE_EXT.ID is '主键ID';
comment on column TABLE_EXT.MAIN_ID is '主表ID';
comment on column TABLE_EXT.FIELD_NAME is '扩展字段名称';
comment on column TABLE_EXT.FIELD_TYPE is '扩展字段类型';
comment on column TABLE_EXT.FIELD_VALUE is '扩展字段值';
comment on column TABLE_EXT.CREATE_ID is '创建人ID';
comment on column TABLE_EXT.CREATE_TIME is '创建时间';
comment on column TABLE_EXT.UPDATE_ID is '修改人ID';
comment on column TABLE_EXT.UPDATE_TIME is '修改时间';
